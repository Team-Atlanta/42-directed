From 03-RESEARCH.md - cpuset parsing (already in runner.sh from Plan 03-01):
```bash
parse_cpuset() {
  local cpuset="$1"
  local cores=()
  IFS=',' read -ra parts <<< "$cpuset"
  for part in "${parts[@]}"; do
    if [[ "$part" =~ ^([0-9]+)-([0-9]+)$ ]]; then
      for ((i=${BASH_REMATCH[1]}; i<=${BASH_REMATCH[2]}; i++)); do
        cores+=("$i")
      done
    else
      cores+=("$part")
    fi
  done
  echo "${cores[@]}"
}
```

```bash
# Background crash copier
(
  while true; do
    find "$SYNC_DIR" -path "*/crashes/*" -type f -newer /tmp/.last_crash_check 2>/dev/null | while read crash; do
      cp "$crash" /fuzzer/crashes/ 2>/dev/null || true
    done
    touch /tmp/.last_crash_check
    sleep 5
  done
) &
```
